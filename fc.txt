    function esEnlaceDeUsuario(enlace) {
        const href = enlace.href || '';
        const onclick = enlace.getAttribute('onclick') || '';
        return href.includes('member.php') ||
               href.includes('/miembros') ||
               href.includes('/perfil') ||
               onclick.includes('member.php') ||
               onclick.includes('/miembros') ||
               onclick.includes('/perfil');
    }

    // Genera una expresión regular para detectar la muletilla al final del mensaje
    function crearRegexMuletilla(frase) {
        let pattern = frase.replace(/[-\/\\^$*+?.()|[\]{}]/g, '\\$&');
        pattern = pattern.replace(/ /g, '(?:\\s|&nbsp;)+');

        if (pattern.endsWith('\\.')) {
            pattern = pattern.slice(0, -2) + '\\.?';
        } else if (pattern.endsWith('\\!')) {
            pattern = pattern.slice(0, -2) + '\\!?';
        } else if (pattern.endsWith('\\?')) {
            pattern = pattern.slice(0, -2) + '\\??';
        }

        const trailingNoise = '((?:\\s|<br\\s*/?>|&nbsp;)*)((?:</[a-z0-9]+>|\\s)*)$';
        return new RegExp(pattern + trailingNoise, 'i');
    }

    function limpiarMuletillas(contenedorPost, autor) {
        const autorLower = autor.toLowerCase();
        if (!(autorLower in muletillasUsuariosLower)) return;

        const fraseABorrar = muletillasUsuariosLower[autorLower];
        const cuerpoMensaje = contenedorPost.querySelector('div[id^="post_message_"], .post_message, .postmessage, [id^="postmessage_"]');

        if (cuerpoMensaje) {
            let html = cuerpoMensaje.innerHTML.trim();
            const regex = crearRegexMuletilla(fraseABorrar);

            if (regex.test(html)) {
                cuerpoMensaje.innerHTML = html.replace(regex, '$2').trim();
            }
        }
    }

    function buscarContenedorPostCompleto(elementoAutor) {
        let el = elementoAutor.parentElement;
        while (el && el !== document.body) {
            const id = el.id || '';

            if (/^post_?\d+$/.test(id)) {
                return el;
            }

            if (el.classList.contains('post-container') ||
                el.classList.contains('post-card') ||
                el.classList.contains('post-wrap') ||
                el.classList.contains('postb') ||
                el.classList.contains('post-block')) {
                return el;
            }

            el = el.parentElement;
        }

        let fallback = elementoAutor.parentElement;
        while (fallback && fallback !== document.body) {
            const tieneCuerpo = fallback.querySelector('div[id^="post_message_"], .post_message, .postmessage, [id^="postmessage_"]');
            if (tieneCuerpo) {
                return fallback;
            }
            fallback = fallback.parentElement;
        }

        return elementoAutor.closest('table') || elementoAutor.closest('div[style*="display"]') || elementoAutor.parentElement;
    }

    function ocultarCitasDeTrolls(contenedorPost) {
        // Encontrar contenedores de citas comunes
        const citas = contenedorPost.querySelectorAll('blockquote, .bbcode_quote, .quote_container, table.alt2, div.bbcode_container');
        citas.forEach(cita => {
            if (cita.hasAttribute('data-fc-quote-processed')) return;

            let autorCitado = null;

            // Buscar etiquetas <b> o <strong> dentro de la cita
            const boldTags = cita.querySelectorAll('strong, b');
            for (let i = 0; i < boldTags.length; i++) {
                const tag = boldTags[i];
                const tagText = tag.textContent.trim();
                if (!tagText) continue;

                // Comprobar si el texto de la cita o del padre del tag indica que es una cita
                const parentText = tag.parentElement ? tag.parentElement.textContent : '';
                const citaText = cita.textContent;

                if (parentText.includes('Iniciado por') || 
                    parentText.includes('Cita de') || 
                    citaText.includes('Iniciado por ' + tagText) || 
                    citaText.includes('Cita de ' + tagText) || 
                    citaText.includes('Escrito originalmente por ' + tagText)) {
                    
                    autorCitado = tagText;
                    break;
                }
            }

            // Fallback: buscar enlaces a perfiles de miembros que puedan estar al inicio de la cita
            if (!autorCitado) {
                const enlaces = cita.querySelectorAll('a');
                for (let i = 0; i < enlaces.length; i++) {
                    const enlace = enlaces[i];
                    if (esEnlaceDeUsuario(enlace)) {
                        const enlaceText = enlace.textContent.trim();
                        if (enlaceText) {
                            if (cita.textContent.includes('Iniciado por') || cita.textContent.includes('Cita de')) {
                                autorCitado = enlaceText;
                                break;
                            }
                        }
                    }
                }
            }

            if (autorCitado) {
                const autorCitadoLower = autorCitado.toLowerCase();
                if (listaNegraLower.has(autorCitadoLower)) {
                    // Marcar como procesado y ocultar
                    cita.setAttribute('data-fc-quote-processed', 'true');
                    cita.style.display = 'none';

                    // Si está dentro de un contenedor bbcode_container superior, ocultamos el contenedor superior también
                    const bbcodeContainer = cita.closest('.bbcode_container');
                    if (bbcodeContainer) {
                        bbcodeContainer.setAttribute('data-fc-quote-processed', 'true');
                        bbcodeContainer.style.display = 'none';
                    }
                }
            }
        });
    }

    function procesarForoCoches() {
        inyectarEstilos();

        // --- 1. MÓVIL: FILTRADO DE HILOS EN LA LISTA DE TEMAS (.threads-list) ---
        const listaHilos = document.querySelector('.threads-list');
        if (listaHilos) {
            const hijosHilos = listaHilos.children;
            for (let i = 0; i < hijosHilos.length; i++) {
                const contenedorHilo = hijosHilos[i];
                if (contenedorHilo.hasAttribute('data-fc-processed')) continue;

                const spansEnHilo = contenedorHilo.querySelectorAll('span');
                let nombreAutor = null;

                for (let j = 0; j < spansEnHilo.length; j++) {
                    const textoSpan = spansEnHilo[j].textContent.trim();
                    if (textoSpan === '@' && spansEnHilo[j + 1]) {
                        nombreAutor = spansEnHilo[j + 1].textContent.trim();
                        break;
                    }
                }

                if (nombreAutor) {
                    contenedorHilo.setAttribute('data-fc-processed', 'true');
                    const nombreLower = nombreAutor.toLowerCase();

                    if (listaNegraLower.has(nombreLower)) {
                        contenedorHilo.classList.add('fc-hilo-peligroso-moderno');
                    } else if (listaAmarilla.has(nombreLower)) {
                        contenedorHilo.classList.add('fc-hilo-sospechoso-moderno');
                    } else if (listaFavoritosLower.has(nombreLower)) {
                        contenedorHilo.classList.add('fc-resaltado-moderno');
                    }
                }
            }
        }

        // --- 2. LAYOUT MODERNO / GAMING: FILTRADO DE HILOS (Divs con Flex) ---
        const enlacesAutoresModernos = document.querySelectorAll('a[href*="showthread.php?p="] span');
        enlacesAutoresModernos.forEach(span => {
            if (span.hasAttribute('data-fc-processed')) return;
            span.setAttribute('data-fc-processed', 'true');

            let texto = span.textContent.trim();
            if (texto.startsWith('@')) {
                const nombreUsuario = texto.substring(1).split(' - ')[0].trim();
                const nombreUsuarioLower = nombreUsuario.toLowerCase();
                const contenedorHiloModerno = span.closest('div[style*="display: flex"][style*="flex-direction: column"]');

                if (contenedorHiloModerno) {
                    if (listaNegraLower.has(nombreUsuarioLower)) {
                        contenedorHiloModerno.classList.add('fc-hilo-peligroso-moderno');
                    } else if (listaAmarilla.has(nombreUsuarioLower)) {
                        contenedorHiloModerno.classList.add('fc-hilo-sospechoso-moderno');
                    } else if (listaFavoritosLower.has(nombreUsuarioLower)) {
                        contenedorHiloModerno.classList.add('fc-resaltado-moderno');
                    }
                }
            }
        });

        // --- 3. PC CLÁSICO: FILTRADO DE HILOS (Tablas - tr[id^="thread_"]) ---
        const filasHilos = document.querySelectorAll('tr[id^="thread_"], #threadslist tr');
        filasHilos.forEach(fila => {
            if (fila.hasAttribute('data-fc-processed')) return;
            if (fila.classList.contains('fc-oculto') || 
                fila.classList.contains('fc-resaltado-rojo') || 
                fila.classList.contains('fc-resaltado-amarillo')) return;

            const celdas = fila.querySelectorAll('td');
            if (celdas.length > 2) {
                let celdaTitulo = null;
                const enlaceTitulo = fila.querySelector('a[id^="thread_title_"]') || fila.querySelector('a[href*="showthread.php"]');
                if (enlaceTitulo) {
                    celdaTitulo = enlaceTitulo.closest('td');
                }

                const celdasABuscar = celdaTitulo ? [celdaTitulo] : Array.from(celdas).slice(1, -1);

                for (let i = 0; i < celdasABuscar.length; i++) {
                    const celda = celdasABuscar[i];
                    const enlace = celda.querySelector('a[href*="member.php?u="], a[href*="/miembros/"], a[href*="/perfil/"]') ||
                                   celda.querySelector('[onclick*="member.php?u="], [onclick*="/miembros/"], [onclick*="/perfil/"]');
                    if (enlace) {
                        const textoAutor = enlace.textContent.trim();
                        const textoAutorLower = textoAutor.toLowerCase();

                        if (listaNegraLower.has(textoAutorLower)) {
                            fila.classList.add('fc-resaltado-rojo');
                            fila.setAttribute('data-fc-processed', 'true');
                            break;
                        } else if (listaAmarilla.has(textoAutorLower)) {
                            fila.classList.add('fc-resaltado-amarillo');
                            fila.setAttribute('data-fc-processed', 'true');
                            break;
                        } else if (listaFavoritosLower.has(textoAutorLower)) {
                            fila.classList.add('fc-resaltado');
                            fila.setAttribute('data-fc-processed', 'true');
                            break;
                        }
                    }
                }
            }
        });

        // --- 4. PROCESAR MENSAJES DENTRO DE LOS HILOS (Tarjetas Completas) ---
        let elementosAutores = Array.from(document.querySelectorAll('.bigusername, .username, .post-author'));

        const todosLosEnlaces = document.querySelectorAll('a, span');
        todosLosEnlaces.forEach(el => {
            if (esEnlaceDeUsuario(el) && el.textContent.trim().length > 0) {
                const enElCuerpo = el.closest('div[id^="post_message_"], .post_message, .postmessage, [id^="postmessage_"], blockquote, .bbcode_quote, .quote_container');
                if (!enElCuerpo) {
                    elementosAutores.push(el);
                }
            }
        });

        elementosAutores.forEach(autorPost => {
            const textoAutor = autorPost.textContent.trim();
            if (!textoAutor) return;

            const contenedorPost = buscarContenedorPostCompleto(autorPost);

            if (contenedorPost) {
                const tieneCuerpo = contenedorPost.querySelector('div[id^="post_message_"], .post_message, .postmessage, [id^="postmessage_"]');
                if (!tieneCuerpo && !contenedorPost.id.startsWith('post')) return;

                if (contenedorPost.hasAttribute('data-fc-processed')) return;
                contenedorPost.setAttribute('data-fc-processed', 'true');

                const textoAutorLower = textoAutor.toLowerCase();

                if (listaNegraLower.has(textoAutorLower)) {
                    contenedorPost.classList.add('fc-oculto');
                    contenedorPost.classList.remove('fc-recuadro-fav', 'fc-sospechoso');

                    const anterior = contenedorPost.previousElementSibling;
                    if (anterior && anterior.tagName.toLowerCase() === 'a' && anterior.name.startsWith('post')) {
                        anterior.classList.add('fc-oculto');
                    }
                } else {
                    limpiarMuletillas(contenedorPost, textoAutor);
                    ocultarCitasDeTrolls(contenedorPost);

                    if (listaFavoritosLower.has(textoAutorLower)) {
                        contenedorPost.classList.add('fc-recuadro-fav');
                    } else if (listaAmarilla.has(textoAutorLower)) {
                        contenedorPost.classList.add('fc-sospechoso');
                    }
                }
            }
        });
    }

    // ==========================================
    // 5. EJECUCIÓN AUTOMÁTICA Y OBSERVER
    // ==========================================

    function iniciarObserver() {
        procesarForoCoches();

        let timeout = null;
        const observer = new MutationObserver(() => {
            if (timeout) clearTimeout(timeout);
            timeout = setTimeout(procesarForoCoches, 50);
        });

        if (document.body) {
            observer.observe(document.body, {
                childList: true,
                subtree: true
            });
        } else {
            const intervalo = setInterval(() => {
                if (document.body) {
                    clearInterval(intervalo);
                    observer.observe(document.body, {
                        childList: true,
                        subtree: true
                    });
                }
            }, 50);
        }
    }

    iniciarObserver();
})();
